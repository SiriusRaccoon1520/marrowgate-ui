// ── helpers ──────────────────────────────────────────────────────────────────

function formatTs(date) {
  const y = date.getFullYear();
  const mo = date.getMonth() + 1;
  const d = date.getDate();
  const h = String(date.getHours()).padStart(2, "0");
  const mi = String(date.getMinutes()).padStart(2, "0");
  return `${y}年${mo}月${d}日 ${h}:${mi}`;
}

function shouldShowTs(messages, idx) {
  if (idx === 0) return true;
  const prev = messages[idx - 1];
  const curr = messages[idx];
  return curr.ts - prev.ts > 60_000;
}

// ── mini markdown renderer ────────────────────────────────────────────────────
function md(raw) {
  if (!raw) return "";
  let s = raw
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;");

  // fenced code
  s = s.replace(/```(\w*)\n?([\s\S]*?)```/g, (_, lang, code) =>
    `<pre class="cb"><code>${code.trimEnd()}</code></pre>`
  );
  // inline code
  s = s.replace(/`([^`]+)`/g, '<code class="ic">$1</code>');
  // bold
  s = s.replace(/\*\*(.+?)\*\*/gs, "<strong>$1</strong>");
  // italic
  s = s.replace(/\*(.+?)\*/g, "<em>$1</em>");
  // h3/h2/h1
  s = s.replace(/^### (.+)$/gm, "<h3>$1</h3>");
  s = s.replace(/^## (.+)$/gm, "<h2>$1</h2>");
  s = s.replace(/^# (.+)$/gm, "<h1>$1</h1>");
  // lists
  s = s.replace(/^[ \t]*[-*] (.+)$/gm, "<li>$1</li>");
  s = s.replace(/(<li>[\s\S]*?<\/li>\n?)+/g, (m) => `<ul>${m}</ul>`);
  // paragraphs
  s = `<p>${s.replace(/\n\n/g, "</p><p>").replace(/\n/g, "<br/>")}</p>`;
  s = s.replace(/<p>(<(?:pre|ul|h[1-3])[^>]*>)/g, "$1");
  s = s.replace(/(<\/(?:pre|ul|h[1-3])>)<\/p>/g, "$1");
  return s;
}

// ── ModelDropdown ─────────────────────────────────────────────────────────────
function ModelDropdown({ model, setModel }) {
  const [open, setOpen] = useState(false);
  const ref = useRef(null);
  const label = MODELS.find((m) => m.id === model)?.label ?? model;

  useEffect(() => {
    const fn = (e) => { if (ref.current && !ref.current.contains(e.target)) setOpen(false); };
    document.addEventListener("mousedown", fn);
    return () => document.removeEventListener("mousedown", fn);
  }, []);

  return (
    <div ref={ref} style={{ position: "relative" }}>
      <button
        onClick={() => setOpen((o) => !o)}
        style={{
          background: "none", border: "none", cursor: "pointer",
          fontSize: 13, color: "#888", fontFamily: "'DM Mono', monospace",
          display: "flex", alignItems: "center", gap: 4, padding: "4px 0",
        }}
      >
        {label}
        <svg width="8" height="5" viewBox="0 0 8 5" fill="none">
          <path d={open ? "M1 4L4 1L7 4" : "M1 1L4 4L7 1"} stroke="#aaa" strokeWidth="1.2" strokeLinecap="round"/>
        </svg>
      </button>
      {open && (
        <div style={{
          position: "absolute", top: "calc(100% + 6px)", left: "50%",
          transform: "translateX(-50%)",
          background: "#fff", border: "1px solid #f0dfe4",
          borderRadius: 10, minWidth: 160, zIndex: 999,
          boxShadow: "0 4px 16px rgba(0,0,0,0.06)",
          overflow: "hidden",
          animation: "ddIn 0.12s ease both",
        }}>
          {MODELS.map((m) => (
            <button key={m.id} onClick={() => { setModel(m.id); setOpen(false); }}
              style={{
                display: "block", width: "100%", padding: "9px 16px",
                textAlign: "left", border: "none", cursor: "pointer",
                background: m.id === model ? "#fdf4f6" : "transparent",
                fontSize: 13, fontFamily: "'DM Mono', monospace",
                color: m.id === model ? "#c05070" : "#555",
                fontWeight: m.id === model ? 600 : 400,
              }}
            >{m.label}</button>
          ))}
        </div>
      )}
    </div>
  );
}

// ── TokenRow ──────────────────────────────────────────────────────────────────
function TokenRow({ meta }) {
  const [show, setShow] = useState(false);
  if (!meta) return null;
  return (
    <div style={{ display: "flex", alignItems: "center", gap: 6, marginTop: 4 }}>
      <button
        onClick={() => setShow((s) => !s)}
        title={show ? "隐藏统计" : "显示统计"}
        style={{
          background: "none", border: "none", cursor: "pointer",
          padding: 0, lineHeight: 1, opacity: show ? 0.7 : 0.3,
          fontSize: 12, transition: "opacity 0.15s",
          color: "#cea0a0",
        }}
      >✰</button>
      {show && (
        <span style={{
          fontSize: 11, color: "#bbb", fontFamily: "'DM Mono', monospace",
          letterSpacing: "0.02em",
        }}>
          ↑ {fmtN(meta.promptTokens)} · ↓ {fmtN(meta.completionTokens)}
          {meta.speed ? ` · ⚡ ${meta.speed} tok/s` : ""}
          {meta.elapsed ? ` · ${meta.elapsed}s` : ""}
        </span>
      )}
    </div>
  );
}

function fmtN(n) {
  if (!n && n !== 0) return "—";
  return n >= 1000 ? (n / 1000).toFixed(1) + "K" : String(n);
}

// ── Bubble ────────────────────────────────────────────────────────────────────
function Bubble({ msg, prevMsg, isStreaming }) {
  const isUser = msg.role === "user";
  const showTs = !prevMsg || (msg.ts - prevMsg.ts > 60_000);

  return (
    <div style={{ marginBottom: 2 }}>
      {showTs && (
        <div style={{
          textAlign: isUser ? "right" : "left",
          fontSize: 11, color: "#bbb",
          padding: isUser ? "12px 4px 4px" : "12px 4px 4px",
          fontFamily: "'DM Mono', monospace",
          letterSpacing: "0.01em",
        }}>
          {formatTs(new Date(msg.ts))}
        </div>
      )}
      <div style={{
        display: "flex",
        justifyContent: isUser ? "flex-end" : "flex-start",
        marginBottom: 6,
      }}>
        <div style={{
          maxWidth: "76%",
          padding: "10px 14px",
          borderRadius: 16,
          background: isUser ? "#e2b0b0" : "#f4e3e3",
          color: "#4e3b3b",
          fontSize: 14.5,
          lineHeight: 1.7,
          wordBreak: "break-word",
          fontFamily: "'Noto Sans SC', 'PingFang SC', sans-serif",
        }}>
          {isUser ? (
            <span style={{ whiteSpace: "pre-wrap" }}>{msg.content}</span>
          ) : (
            <span
              className="md-render"
              dangerouslySetInnerHTML={{ __html: md(msg.content) }}
            />
          )}
          {isStreaming && (
            <span style={{
              display: "inline-block", width: 1.5, height: "1em",
              background: "#c06080", marginLeft: 2,
              verticalAlign: "text-bottom", borderRadius: 1,
              animation: "blink 0.8s step-start infinite",
            }}/>
          )}
        </div>
      </div>
      {!isUser && <TokenRow meta={msg.meta} />}
    </div>
  );
}
